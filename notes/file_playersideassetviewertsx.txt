/* ── PlayerSideAssetViewer ─────────────────────────────────────
 * Lightweight overlay that appears on player-facing screens when
 * the DM shares an asset. Shows the shared image with a brief
 * animation, plus Accept/Decline buttons for inventory deposit.
 *
 * DESIGN: Minimal glassmorphism card that auto-dismisses.
 * Positioned bottom-center with slide-up animation.
 * ─────────────────────────────────────────────────────────────── */

import { useState, useEffect } from "react";
import { useSharedAssetStore } from "@/stores/sharedAssetStore";
import { useCampaignStore } from "@/stores/campaignStore";
import { useAuthStore } from "@/stores/authStore";
import { Button } from "@/components/ui/Button";

interface PlayerSideAssetViewerProps {
  /** Player character ID (for deposit targeting) */
  characterId: string;
  /** Player name (for display) */
  playerName: string;
  /** Character name */
  characterName: string;
}

export function PlayerSideAssetViewer({ characterId, playerName, characterName }: PlayerSideAssetViewerProps) {
  const { currentAsset, pendingDeposits, respondToDeposit } = useSharedAssetStore();
  const updatePlayerCharacter = useCampaignStore((s) => s.updatePlayerCharacter);
  const characters = useCampaignStore((s) => s.characters);
  const [dismissed, setDismissed] = useState(false);
  const [showDepositPrompt, setShowDepositPrompt] = useState(false);

  const pendingDeposit = pendingDeposits.find(
    (d) => d.characterId === characterId && d.status === "pending"
  );

  // Show deposit prompt when a pending deposit arrives
  useEffect(() => {
    if (pendingDeposit) {
      setShowDepositPrompt(true);
      setDismissed(false);
    }
  }, [pendingDeposit?.requestedAt]);

  // Auto-dismiss after 10 seconds if not interacted with
  useEffect(() => {
    if (!currentAsset || dismissed) return;
    const timer = setTimeout(() => setDismissed(true), 10000);
    return () => clearTimeout(timer);
  }, [currentAsset, dismissed]);

  if (!currentAsset || dismissed) return null;

  /* ── Handle Accept Deposit ── */
  const handleAccept = () => {
    const character = characters.find((c) => c.id === characterId);
    if (!character) return;

    const newItem = {
      id: `shared_${Date.now()}_${Math.random().toString(36).slice(2, 8)}`,
      name: currentAsset.label || "Shared Image",
      description: currentAsset.dmNote || `Shared by DM on ${new Date(currentAsset.sharedAt).toLocaleDateString()}`,
      imageUrl: currentAsset.imageUrl,
      quantity: 1,
      weight: 0,
      type: "quest" as const,
      rarity: "common" as const,
      attunement: false,
      equipped: false,
    };

    const updatedInventory = [...(character.inventory || []), newItem];
    updatePlayerCharacter(characterId, { inventory: updatedInventory });
    respondToDeposit(characterId, true);
    setShowDepositPrompt(false);
    setDismissed(true);
  };

  /* ── Handle Decline Deposit ── */
  const handleDecline = () => {
    respondToDeposit(characterId, false);
    setShowDepositPrompt(false);
  };

  return (
    <div className="fixed bottom-4 left-1/2 -translate-x-1/2 z-50 animate-slide-up">
      <div className="rounded-xl border border-surface-600/60 bg-surface-850/95 shadow-2xl shadow-black/30 backdrop-blur-xl p-3 w-[320px] max-w-[calc(100vw-32px)]">
        {/* Asset Display */}
        <div className="flex gap-3">
          {currentAsset.imageUrl && (
            <div className="relative w-20 h-20 shrink-0 rounded-lg overflow-hidden border border-surface-700">
              <img
                src={currentAsset.imageUrl}
                alt={currentAsset.label}
                className="w-full h-full object-cover"
              />
            </div>
          )}
          <div className="flex-1 min-w-0">
            <div className="flex items-center gap-2">
              <span className="text-sm">📦</span>
              <span className="text-xs font-semibold text-surface-200">
                DM Shared an Asset
              </span>
            </div>
            <p className="text-[11px] font-medium text-surface-300 mt-1 truncate">
              {currentAsset.label}
            </p>
            {currentAsset.dmNote && (
              <p className="text-[9px] text-surface-400 mt-0.5 line-clamp-2">
                {currentAsset.dmNote}
              </p>
            )}

            {/* Deposit Prompt */}
            {showDepositPrompt && pendingDeposit && (
              <div className="mt-2 space-y-1.5">
                <p className="text-[10px] text-accent-400 font-medium">
                  DM wants to deposit this to your inventory
                </p>
                <div className="flex gap-2">
                  <Button size="xs" onClick={handleAccept}>
                    ✅ Accept
                  </Button>
                  <Button size="xs" variant="ghost" onClick={handleDecline}>
                    ❌ Decline
                  </Button>
                </div>
              </div>
            )}

            {/* Dismiss */}
            {!showDepositPrompt && (
              <button
                onClick={() => setDismissed(true)}
                className="mt-1.5 text-[9px] text-surface-500 hover:text-surface-300"
              >
                Dismiss
              </button>
            )}
          </div>
        </div>
      </div>
    </div>
  );
}
