```typescript
/* ══════════════════════════════════════════════════════════════

/* ── Homebrew Species ───────────────────────────────────────── */

export interface HomebrewSpecies {
  id: string;
  name: string;                     // e.g. "Aarakocra", "Dragonborn"
  description: string;              // Full racial description
  flavorText?: string;              // RP fluff
  abilityScoreBonuses: {            // e.g. { strength: 2, charisma: 1 }
    strength?: number;
    dexterity?: number;
    constitution?: number;
    intelligence?: number;
    wisdom?: number;
    charisma?: number;
  };
  size: "tiny" | "small" | "medium" | "large";
  speed: number;                    // base walking speed in feet
  speedTypes?: {                    // alternative movement types
    fly?: number;
    swim?: number;
    climb?: number;
    burrow?: number;
  };
  darkvision?: number;              // feet, if applicable
  racialTraits: {
    name: string;
    description: string;
    mechanical: string;             // exact game effect
  }[];
  languages: string[];              // e.g. ["Common", "Draconic"]
  age?: string;                     // narrative description
  alignment?: string;               // typical alignment tendencies
  tags: string[];
  source: string;                   // e.g. "Player's Handbook", "Homebrew"
  isHomebrew: boolean;
  createdAt: number;
  updatedAt: number;
}

/* ── Homebrew Class ─────────────────────────────────────────── */

export interface HomebrewClass {
  id: string;
  name: string;                     // e.g. "Blood Hunter"
  description: string;
  flavorText?: string;
  hitDice: string;                  // e.g. "d10"
  primaryAbility: string;           // e.g. "Strength", "Dexterity"
  savingThrows: string[];           // e.g. ["Strength", "Constitution"]
  armorProficiencies: string[];
  weaponProficiencies: string[];
  toolProficiencies: string[];
  skillChoices: number;             // how many skills to choose
  availableSkills: string[];        // list of eligible skills
  spellcastingAbility?: string;     // e.g. "Intelligence" — null if non-spellcaster
  subclassLabel: string;            // e.g. "Martial Archetype", "College"
  classFeatures: {
    level: number;
    name: string;
    description: string;
    mechanical: string;
  }[];
  multiclassRequirements?: {
    ability: string;
    minimum: number;
  };
  source: string;
  isHomebrew: boolean;
  tags: string[];
  createdAt: number;
  updatedAt: number;
}

/* ── Homebrew Subclass ──────────────────────────────────────── */

export interface HomebrewSubclass {
  id: string;
  name: string;                     // e.g. "Path of the Giant"
  className: string;                // parent class e.g. "Barbarian"
  description: string;
  flavorText?: string;
  spellcastingAbility?: string;     // if subclass grants spellcasting
  subclassFeatures: {
    level: number;
    name: string;
    description: string;
    mechanical: string;
  }[];
  bonusSpells?: {                   // if subclass grants extra known spells
    level: number;
    spellName: string;
  }[];
  source: string;
  isHomebrew: boolean;
  tags: string[];
  createdAt: number;
  updatedAt: number;
}

/* ── Homebrew Currency ──────────────────────────────────────── */

export interface HomebrewCurrency {
  id: string;
  name: string;                     // e.g. "Platinum Piece", "Gold Piece"
  pluralName: string;               // e.g. "Platinum Pieces"
  abbreviation: string;             // e.g. "PP", "GP"
  symbol: string;                   // e.g. "₽", "Ⓖ" — emoji or unicode
  conversionToGold: number;         // 1 unit = X gold (e.g. 10 for PP, 0.01 for CP)
  color: string;                    // Tailwind color class for display
  description?: string;
  sortOrder: number;                // for display ordering
  source: string;
  isHomebrew: boolean;
  tags: string[];
  createdAt: number;
  updatedAt: number;
}

/* ── Extended Library ───────────────────────────────────────── */
// Extend the existing HomebrewLibrary interface
export interface HomebrewLibraryExtended {
  species: HomebrewSpecies[];
  classes: HomebrewClass[];
  subclasses: HomebrewSubclass[];
  currencies: HomebrewCurrency[];
}
```

## 3. DEBOUNCED AUTO-SAVING DM NOTES EDITOR
### New component at `vtt/src/components/journal/DmNotesEditor.tsx`

```
Props:
  - noteId: string
  - initialContent: string
  - onSave: (content: string) => void
  - placeholder?: string

Behavior:
  - Renders a textarea with premium glassmorphism styling
  - On input: stores to local state immediately (snappy UX)
  - Debounce 800ms: triggers onSave callback
  - Visual indicator: "Saving..." → "Saved" → "Changes saved"
  - Auto-saves on blur (immediate, no debounce wait)
  - Persists to localStorage as fallback if Firebase is offline
  - Re-inflates from Firebase on mount (via noteId)

State machine:
  idle → typing → saving → saved → idle
  Error state: "Failed to save" with retry button

Integration:
  - DmDashboard renders DmNotesEditor as a full-width card
  - DM Journal page uses it for individual entry editing
```
